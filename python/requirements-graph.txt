# graph_universe_sync deps.
#
# Reads from MariaDB (pymysql), writes to Neo4j via the official driver,
# and emits outbox ULIDs using python-ulid. Split from the SDE importer
# requirements so the two one-shot images stay lean.
neo4j~=5.26
pymysql~=1.1
python-ulid~=3.0
# python-ulid 3.0.x has an unconditional `from typing_extensions import Self`
# but only declares typing_extensions as a dep on Python < 3.11 — fine on
# 3.10, blows up at import time on 3.12 (our base image). Pin it ourselves
# until python-ulid drops the unconditional import. See:
#   https://github.com/mdomke/python-ulid/issues — search "typing_extensions".
typing_extensions>=4.7
